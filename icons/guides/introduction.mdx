export const icons = [
  {
    name: "ls:livesession",
    label: "livesession"
  },
  {
    name: "ls:play",
    label: "play"
  },
  {
    name: "ls:metrics",
    label: "metrics"
  }
];

export function IconsGrid({ components }) {
  const { Icon = () => null } = components ?? {};

  return (
    <div data-content-element className="icons-grid">
      {icons.map((icon) => (
        <button key={icon.name} className="icon-preview">
          <Icon name={icon.name} />
          <span>{icon.label}</span>
        </button>
      ))}
    </div>
  );
}
